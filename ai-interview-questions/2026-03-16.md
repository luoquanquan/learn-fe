### 四、常见误区

❌ **错误做法**:
- 一次性展示所有安全警告，用户产生疲劳
- 强制用户观看长篇教程才能使用
- 专业术语过多（如"非对称加密"）

✅ **正确做法**:
- 在相关场景下提示对应风险
- 提供"快速开始"选项，复杂功能后续学习
- 用类比解释概念（如"助记词就像保险柜的万能钥匙"）

class IPFSResolver {
  private gatewayIndex = 0;
  
  resolve(url: string): string {
    if (!url.startsWith('ipfs://')) return url;

}

### 三、元数据解析流程

```typescript
class MetadataService {
  private cache = new Map<string, NFTMetadata>();
  
  async fetchMetadata(tokenURI: string): Promise<NFTMetadata> {
    // 1. 检查缓存
    if (this.cache.has(tokenURI)) {
      return this.cache.get(tokenURI)!;
    }
    
    // 2. 解析 URI 类型
    const url = this.resolveURI(tokenURI);
    
    // 3. 获取数据
    const response = await fetch(url, {
      headers: { 'Accept': 'application/json' },
    });
    
    // 4. 验证和解析
    const raw = await response.json();
    const metadata = this.validateMetadata(raw);
    
    // 5. 缓存结果
    this.cache.set(tokenURI, metadata);
    
    return metadata;
  }
  
  private resolveURI(uri: string): string {
    if (uri.startsWith('ipfs://')) {
      return uri.replace('ipfs://', 'https://cloudflare-ipfs.com/ipfs/');
    }
    if (uri.startsWith('ar://')) {
      return uri.replace('ar://', 'https://arweave.net/');
    }
    return uri;
  }
  
  private validateMetadata(raw: any): NFTMetadata {
    // 使用 zod 进行运行时类型验证
    const schema = z.object({
      name: z.string(),
      description: z.string().optional(),
      image: z.string(),
      attributes: z.array(z.object({
        trait_type: z.string(),
        value: z.union([z.string(), z.number()]),
      })).optional(),
    });
    
    return schema.parse(raw);
  }
}
```

**2. Injected Provider - EIP-1193 标准实现**
```javascript
// injected-provider.js
class BinanceWalletProvider extends EventEmitter {
  constructor() {
    super();
    this.chainId = null;
    this.accounts = [];
    this.isConnected = false;
  }
  
  // EIP-1193 标准方法
  async request(args) {
    return new Promise((resolve, reject) => {
      const id = Date.now() + Math.random();
      
      // 发送请求到 content script
      window.postMessage({
        type: 'WALLET_REQUEST',
        payload: { id, ...args },
      }, '*');
      
      // 监听响应
      const handler = (event) => {
        if (event.data.type !== 'WALLET_RESPONSE') return;
        if (event.data.payload.id !== id) return;
        
        window.removeEventListener('message', handler);
        
        if (event.data.payload.error) {
          reject(new Error(event.data.payload.error));
        } else {
          resolve(event.data.payload.result);
        }
      };
      
      window.addEventListener('message', handler);
    });
  }
  
  // 断开连接
  disconnect() {
    this.isConnected = false;
    this.accounts = [];
    this.emit('disconnect', { code: 1000, message: 'User disconnected' });
  }
}

// 注入到 window
window.binanceWallet = new BinanceWalletProvider();
window.dispatchEvent(new Event('binanceWallet#initialized'));
```

### 二、安全集成要点

**1. 合约安全验证**
```typescript
interface ProtocolSecurity {
  audited: boolean;
  auditor: string;
  auditReportUrl: string;
  bugBounty: boolean;
  tvl: bigint; // 总锁仓价值
  age: number; // 上线天数
  incidents: SecurityIncident[];
}

const PROTOCOL_WHITELIST: Record<string, ProtocolSecurity> = {
  '0x...pancakeSwap': {
    audited: true,
    auditor: 'Certik, SlowMist',
    auditReportUrl: 'https://...',
    bugBounty: true,
    tvl: 5000000000n, // $5B
    age: 1000,
    incidents: [],
  },
  // ...
};

// 风险评分算法
function calculateRiskScore(protocol: ProtocolSecurity): number {
  let score = 100;
  
  if (!protocol.audited) score -= 30;

    { name: 'provider', getGasPrice: () => this.getFromProvider() },
    { name: 'etherscan', getGasPrice: () => this.getFromEtherscan() },
    { name: 'blocknative', getGasPrice: () => this.getFromBlocknative() },

  async estimate(): Promise<GasEstimate> {
    // 并行获取多个数据源
    const results = await Promise.allSettled(
      this.oracles.map(o => o.getGasPrice())
    );
    
    const estimates = results
      .filter((r): r is PromiseFulfilledResult<GasEstimate> => r.status === 'fulfilled')
      .map(r => r.value);
    
    return {
      slow: this.percentile(estimates.map(e => e.slow), 0.3),
      standard: this.percentile(estimates.map(e => e.standard), 0.5),
      fast: this.percentile(estimates.map(e => e.fast), 0.8),
    };
  }
  
  private percentile(values: number[], p: number): number {
    const sorted = [...values].sort((a, b) => a - b);
    const index = Math.ceil(sorted.length * p) - 1;
    return sorted[Math.max(0, index)];
  }
}

interface NetworkCongestion {
  level: 'low' | 'medium' | 'high' | 'critical';
  baseFee: bigint;
  pendingTxCount: number;
  blockUtilization: number; // 0-1

class CongestionDetector {
  async checkCongestion(): Promise<NetworkCongestion> {
    const [feeData, block, pendingTxs] = await Promise.all([
      provider.getFeeData(),
      provider.getBlock('latest'),
      provider.send('txpool_status', []),
    ]);
    
    const baseFee = feeData.maxFeePerGas! - feeData.maxPriorityFeePerGas!;
    const gasUsed = BigInt(block.gasUsed);
    const gasLimit = BigInt(block.gasLimit);
    const utilization = Number(gasUsed) / Number(gasLimit);
    
    // 拥堵分级
    let level: NetworkCongestion['level'];
    if (utilization < 0.5 && baseFee < parseGwei('20')) {
      level = 'low';
    } else if (utilization < 0.8 && baseFee < parseGwei('50')) {
      level = 'medium';
    } else if (utilization < 0.95 && baseFee < parseGwei('100')) {
      level = 'high';
    } else {
      level = 'critical';
    }
    
    return {
      level,
      baseFee,
      pendingTxCount: parseInt(pendingTxs.pending, 16),
      blockUtilization: utilization,
      estimatedWaitTime: this.estimateWaitTime(level),
    };
  }
  
  private estimateWaitTime(level: string): number {
    const times = { low: 15, medium: 60, high: 300, critical: 1800 };
    return times[level] || 60;
  }
}

class TransactionMonitor {
  private transactions: Map<string, TransactionStatus> = new Map();
  
  async monitorTransaction(txHash: string, nonce: number) {
    const startTime = Date.now();
    
    while (Date.now() - startTime < 30 * 60 * 1000) { // 监控30分钟
      const receipt = await provider.getTransactionReceipt(txHash);
      
      if (receipt) {
        this.updateStatus(txHash, { status: 'confirmed', receipt });
        return;
      }
      
      const tx = await provider.getTransaction(txHash);
      if (!tx) {
        // 交易从 mempool 消失，可能被替换或丢弃
        this.updateStatus(txHash, { status: 'dropped' });
        return;
      }
      
      // 检查是否 stuck
      const currentNonce = await provider.getTransactionCount(tx.from, 'latest');
      if (nonce < currentNonce) {
        this.updateStatus(txHash, { status: 'superseded' });
        return;
      }
      
      // 检查等待时间
      const waitMinutes = (Date.now() - startTime) / 60000;
      if (waitMinutes > 5) {
        this.updateStatus(txHash, { 
          status: 'stuck',
          options: ['speedup', 'cancel'],
        });
      }
      
      await sleep(15000); // 15秒检查一次
    }
  }

class TransactionAccelerator {
  // 加速交易（替换为更高 Gas 的交易）
  async speedUpTransaction(originalTx: TransactionResponse): Promise<string> {
    const feeData = await provider.getFeeData();
    
    // 新 Gas 价格必须比原交易高至少 10%
    const newMaxFee = (originalTx.maxFeePerGas! * 110n) / 100n;
    const newPriorityFee = (originalTx.maxPriorityFeePerGas! * 110n) / 100n;
    
    // 确保新价格不低于当前网络推荐
    const finalMaxFee = max(newMaxFee, feeData.maxFeePerGas!);
    const finalPriorityFee = max(newPriorityFee, feeData.maxPriorityFeePerGas!);
    
    const newTx = {
      to: originalTx.to,
      from: originalTx.from,
      value: originalTx.value,
      data: originalTx.data,
      nonce: originalTx.nonce, // 关键：相同 nonce
      gasLimit: originalTx.gasLimit,
      maxFeePerGas: finalMaxFee,
      maxPriorityFeePerGas: finalPriorityFee,
      type: 2, // EIP-1559
      chainId: originalTx.chainId,
    };
    
    const signed = await signer.signTransaction(newTx);
    const newTxHash = await provider.send('eth_sendRawTransaction', [signed]);
    
    return newTxHash;
  }
  
  // 取消交易（发送 0 ETH 给自己）
  async cancelTransaction(originalTx: TransactionResponse): Promise<string> {
    const feeData = await provider.getFeeData();
    
    const cancelTx = {
      to: originalTx.from, // 发送给自己
      from: originalTx.from,
      value: 0n,
      nonce: originalTx.nonce, // 相同 nonce
      gasLimit: 21000n, // 标准转账 Gas
      maxFeePerGas: (feeData.maxFeePerGas! * 120n) / 100n,
      maxPriorityFeePerGas: (feeData.maxPriorityFeePerGas! * 120n) / 100n,
      type: 2,
      chainId: originalTx.chainId,
    };
    
    const signed = await signer.signTransaction(cancelTx);
    return provider.send('eth_sendRawTransaction', [signed]);
  }

class SmartPoller {
  private attemptCount = 0;
  private baseInterval = 5000; // 5秒
  private maxInterval = 60000; // 60秒
  
  async pollWithBackoff(txHash: string): Promise<TransactionReceipt | null> {
    const receipt = await provider.getTransactionReceipt(txHash);
    
    if (receipt) {
      this.attemptCount = 0;
      return receipt;
    }
    
    // 指数退避
    this.attemptCount++;
    const delay = Math.min(
      this.baseInterval * Math.pow(1.5, this.attemptCount),
      this.maxInterval
    );
    
    await sleep(delay);
    return this.pollWithBackoff(txHash);
  }
  
  // 优先级轮询（新交易更频繁）
  getPollInterval(submittedAt: number): number {
    const age = Date.now() - submittedAt;
    
    if (age < 60000) return 3000;      // < 1分钟: 3秒
    if (age < 300000) return 10000;    // < 5分钟: 10秒
    if (age < 600000) return 30000;    // < 10分钟: 30秒
    return 60000;                       // > 10分钟: 60秒
  }
}

class DCAService {
  async executeStrategy(strategy: DCAStrategy) {
    // 1. 检查条件
    const shouldExecute = await this.checkSchedule(strategy);
    if (!shouldExecute) return;
    
    // 2. 风控检查
    const riskCheck = await this.riskAssessment(strategy);
    if (!riskCheck.approved) {
      await this.notifyUser(strategy, '风控拦截', riskCheck.reason);
      return;
    }
    
    // 3. 获取报价
    const quote = await this.getSwapQuote({
      from: strategy.config.quoteToken,
      to: strategy.config.token,
      amount: strategy.config.amount,
    });
    
    // 4. 用户确认（可选，取决于配置）
    if (strategy.config.requireConfirmation) {
      const approved = await this.requestUserApproval(strategy, quote);
      if (!approved) return;
    }
    
    // 5. 执行交易
    const tx = await this.executeSwap(quote);
    
    // 6. 记录和通知
    await this.recordExecution(strategy, tx);
    await this.notifyUser(strategy, '定投执行成功', {
      amount: strategy.config.amount,
      received: quote.outputAmount,
      price: quote.price,
    });
  }
  
  private async checkSchedule(strategy: DCAStrategy): Promise<boolean> {
    const now = new Date();
    const last = strategy.stats.lastExecution;
    
    switch (strategy.config.frequency) {
      case 'daily':
        return !last || now.getDate() !== last.getDate();
      case 'weekly':
        return now.getDay() === strategy.config.dayOfWeek &&
               (!last || now.getTime() - last.getTime() > 6 * 24 * 60 * 60 * 1000);
      case 'monthly':
        return now.getDate() === strategy.config.dayOfMonth &&
               (!last || now.getMonth() !== last.getMonth());
    }
  }
}

class RiskController {
  async assessRisk(
    userId: string,
    strategy: Strategy,
    proposedTx: Transaction,
  ): Promise<RiskAssessment> {
    const config = await this.getUserRiskConfig(userId);
    const checks: RiskCheck[] = [];
    
    // 1. 额度检查
    const todaySpent = await this.getTodaySpent(userId);
    if (todaySpent + proposedTx.amount > config.dailyLimit) {
      checks.push({
        type: 'limit_exceeded',
        severity: 'high',
        message: '超过每日自动交易限额',
      });
    }
    
    // 2. 频率检查
    const recentTxs = await this.getRecentTransactions(userId, '1h');
    if (recentTxs.length >= config.maxTxPerHour) {
      checks.push({
        type: 'frequency_exceeded',
        severity: 'medium',
        message: '交易频率过高，请稍后再试',
      });
    }
    
    // 3. 异常检测
    if (await this.detectAnomaly(proposedTx)) {
      checks.push({
        type: 'anomaly_detected',
        severity: 'high',
        message: '检测到异常交易模式，需要人工确认',
      });
    }
    
    // 4. 价格冲击检查
    const priceImpact = await this.calculatePriceImpact(proposedTx);
    if (priceImpact > 5) {
      checks.push({
        type: 'high_price_impact',
        severity: 'medium',
        message: `价格影响 ${priceImpact.toFixed(2)}%，可能造成较大滑点`,
      });
    }
    
    // 综合评估
    const hasBlocking = checks.some(c => c.severity === 'high');
    return {
      approved: !hasBlocking,
      checks,
      requiredConfirmations: checks.filter(c => c.severity === 'medium').length,
    };
  }
  
  // AI 异常检测
  private async detectAnomaly(tx: Transaction): Promise<boolean> {
    // 使用历史交易数据训练模型
    const userPattern = await this.getUserPattern(tx.from);
    
    // 检查是否偏离正常模式
    const anomalies = [
      tx.amount > userPattern.avgAmount * 10,      // 金额异常大
      tx.to !== userPattern.commonCounterparties,  // 对手方异常
      tx.token !== userPattern.commonTokens,       // 代币异常
      tx.time.getHours() !== userPattern.activeHours, // 时间异常
    ];
    
    return anomalies.filter(Boolean).length >= 2;
  }
}

class AddressClustering:
    """基于启发式规则的地址聚类"""
    
    def __init__(self):
        self.clusters = {}  # cluster_id -> set(addresses)
        self.address_to_cluster = {}  # address -> cluster_id
    
    def same_input_clustering(self, transactions):
        """
        共同输入聚类（多对一交易）
        如果多个地址在一笔交易中作为输入，它们可能属于同一实体
        """
        for tx in transactions:
            inputs = tx['inputs']
            if len(inputs) > 1:
                addresses = [inp['address'] for inp in inputs]
                self.merge_addresses(addresses)
    
    def one_time_change_clustering(self, transactions):
        """
        找零地址聚类
        假设：一笔交易中的新输出地址（未使用过）可能是找零地址
        """
        for tx in transactions:
            outputs = tx['outputs']
            # 识别新地址（首次出现）
            new_addresses = [
                out['address'] for out in outputs 
                if self.is_new_address(out['address'])
            ]
            if len(new_addresses) == 1:
                # 可能是找零地址，与输入地址聚类
                input_addresses = [inp['address'] for inp in tx['inputs']]
                self.merge_addresses(input_addresses + new_addresses)
    
    def temporal_analysis(self, transactions):
        """
        时间分析聚类
        如果地址 A 和 B 总是同时活跃，可能属于同一实体
        """
        address_activity = defaultdict(list)
        
        for tx in transactions:
            timestamp = tx['timestamp']
            for addr in tx['all_addresses']:
                address_activity[addr].append(timestamp)
        
        # 计算地址对的时间相关性
        for addr1, addr2 in combinations(address_activity.keys(), 2):
            correlation = self.time_correlation(
                address_activity[addr1],
                address_activity[addr2]
            )
            if correlation > 0.8:
                self.merge_addresses([addr1, addr2])

        optimizer = torch.optim.Adam(self.model.parameters(), lr=0.01)
        
        for epoch in range(epochs):
            optimizer.zero_grad()
            loss = self.model.loss()
            loss.backward()
            optimizer.step()
    
    def get_embeddings(self):
        """获取所有节点的嵌入向量"""
        self.model.eval()
        with torch.no_grad():
            embeddings = self.model()
        return embeddings
    
    def cluster_addresses(self, n_clusters=100):

        embeddings = self.get_embeddings().numpy()

class StatisticalAnomalyDetector:
    """基于统计方法的异常检测"""
    
    def __init__(self):
        self.user_profiles = {}  # address -> normal behavior profile
    
    def build_profile(self, address, transactions):
        """构建用户正常行为画像"""
        amounts = [tx['amount'] for tx in transactions]
        times = [tx['timestamp'] for tx in transactions]
        
        profile = {
            'avg_amount': np.mean(amounts),
            'std_amount': np.std(amounts),
            'avg_frequency': len(transactions) / (max(times) - min(times)),
            'usual_counterparties': self.get_top_counterparties(transactions),
            'active_hours': self.get_active_hours(transactions),
            'usual_tokens': self.get_usual_tokens(transactions),
        }
        
        self.user_profiles[address] = profile
    
    def detect_anomaly(self, address, new_transaction):
        """检测交易是否异常"""
        profile = self.user_profiles.get(address)
        if not profile:
            return {'is_anomaly': False, 'reason': 'no_profile'}
        
        anomalies = []
        
        # 金额异常
        amount_zscore = (
            new_transaction['amount'] - profile['avg_amount']
        ) / profile['std_amount']
        if abs(amount_zscore) > 3:
            anomalies.append({
                'type': 'amount_anomaly',
                'zscore': amount_zscore,
                'severity': 'high' if abs(amount_zscore) > 5 else 'medium'
            })
        
        # 对手方异常
        if new_transaction['to'] not in profile['usual_counterparties']:
            anomalies.append({
                'type': 'new_counterparty',
                'severity': 'low'
            })
        
        # 时间异常
        hour = datetime.fromtimestamp(new_transaction['timestamp']).hour
        if hour not in profile['active_hours']:
            anomalies.append({
                'type': 'unusual_time',
                'severity': 'low'
            })
        
        return {
            'is_anomaly': len(anomalies) > 0,
            'anomalies': anomalies,
            'risk_score': self.calculate_risk_score(anomalies)
        }

from torch_geometric.nn import GCNConv, GATConv

class FraudDetectionGNN(nn.Module):
    """用于欺诈检测的图神经网络"""
    
    def __init__(self, in_channels, hidden_channels, out_channels):
        super().__init__()
        
        # GCN 层
        self.conv1 = GCNConv(in_channels, hidden_channels)
        self.conv2 = GCNConv(hidden_channels, hidden_channels)
        
        # 注意力层
        self.attention = GATConv(hidden_channels, hidden_channels, heads=4)
        
        # 分类层
        self.classifier = nn.Sequential(
            nn.Linear(hidden_channels * 4, hidden_channels),
            nn.ReLU(),
            nn.Dropout(0.5),
            nn.Linear(hidden_channels, out_channels)
        )
    
    def forward(self, x, edge_index, edge_attr):
        # 图卷积
        x = self.conv1(x, edge_index)
        x = F.relu(x)
        x = F.dropout(x, p=0.5, training=self.training)
        
        x = self.conv2(x, edge_index)
        x = F.relu(x)
        
        # 注意力机制
        x = self.attention(x, edge_index, edge_attr)
        x = F.relu(x)
        
        # 分类
        out = self.classifier(x)
        return F.log_softmax(out, dim=1)

            'amount_max_7d': df.last(window)['amount'].max(),
            'unique_counterparties_7d': df.last(window)['to'].nunique(),
        }
        return features
```

```python
class RealTimeFraudDetection:
    """实时欺诈检测系统"""
    
    def __init__(self):
        self.anomaly_detector = IsolationForestDetector()
        self.gnn_model = FraudDetectionGNN(128, 64, 2)
        self.known_fraud_addresses = self.load_fraud_database()
    
    async def process_transaction(self, tx):
        """处理新交易"""
        # 1. 检查已知欺诈地址
        if tx['from'] in self.known_fraud_addresses or \
           tx['to'] in self.known_fraud_addresses:
            return self.create_alert(tx, 'known_fraud_address', 'critical')
        
        # 2. 规则引擎快速筛查
        rule_result = self.rule_based_check(tx)
        if rule_result['blocked']:
            return self.create_alert(tx, rule_result['reason'], 'high')
        
        # 3. 异常检测模型
        anomaly_result = self.anomaly_detector.predict(tx)
        if anomaly_result['is_anomaly'] and anomaly_result['confidence'] > 0.7:
            return self.create_alert(tx, 'anomaly_detected', 'medium')
        
        # 4. 图神经网络预测
        gnn_result = await self.gnn_predict(tx)
        if gnn_result['fraud_probability'] > 0.8:
            return self.create_alert(tx, 'gnn_fraud_prediction', 'high')
        
        return {'status': 'safe'}
    
    def rule_based_check(self, tx):
        """基于规则的快速检查"""
        rules = [
            # 金额异常大
            lambda t: t['amount'] > 1000 and {'blocked': True, 'reason': 'large_amount'},
            
            # 混币器地址
            lambda t: t['to'] in TORNADO_CASH_ADDRESSES and {'blocked': True, 'reason': 'mixer_detected'},
            
            # 闪电贷攻击特征
            lambda t: self.is_flash_loan_pattern(t) and {'blocked': True, 'reason': 'flash_loan_pattern'},
            
            # 新地址大额转账
            lambda t: self.is_new_address(t['from']) and t['amount'] > 100 and {'blocked': True, 'reason': 'new_address_large_tx'},
        ]
        
        for rule in rules:
            result = rule(tx)
            if result and result.get('blocked'):
                return result
        
        return {'blocked': False}
```
