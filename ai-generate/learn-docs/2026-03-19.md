## 1. TypeScript 类型体操：实现类型安全的 EventEmitter

**考点**: TypeScript、泛型、条件类型、映射类型、类型推断

**答案要点**:

### 核心原理解释

EventEmitter 是前端开发中常用的模式，但在 TypeScript 中实现类型安全的事件系统需要深入理解泛型和映射类型。

关键点：
1. **泛型约束**：使用 `extends` 限制事件名称和回调参数类型
2. **映射类型**：根据事件名称映射到对应的回调参数类型
3. **类型推断**：让 TypeScript 自动推断 emit 时的参数类型
4. **分布式条件类型**：处理联合类型的分发

### 代码示例

```typescript
// 类型安全的事件发射器实现
type EventMap = {
  [key: string]: (...args: any[]) => void;
};

class TypedEventEmitter<Events extends EventMap> {
  private listeners: { [K in keyof Events]?: Events[K][] } = {};

  // on 方法：类型安全的监听注册
  on<K extends keyof Events>(
    event: K,
    listener: Events[K]
  ): this {
    if (!this.listeners[event]) {
      this.listeners[event] = [];
    }
    this.listeners[event]!.push(listener);
    return this;
  }

  // emit 方法：类型安全的事件触发
  emit<K extends keyof Events>(
    event: K,
    ...args: Parameters<Events[K]>
  ): boolean {
    const handlers = this.listeners[event];
    if (!handlers || handlers.length === 0) {
      return false;
    }
    
    handlers.forEach(handler => {
      handler(...args);
    });
    return true;
  }

  // off 方法：移除监听
  off<K extends keyof Events>(
    event: K,
    listener: Events[K]
  ): this {
    const handlers = this.listeners[event];
    if (handlers) {
      const index = handlers.indexOf(listener);
      if (index > -1) {
        handlers.splice(index, 1);
      }
    }
    return this;
  }

  // once 方法：只执行一次
  once<K extends keyof Events>(
    event: K,
    listener: Events[K]
  ): this {
    const onceListener = (...args: Parameters<Events[K]>) => {
      this.off(event, onceListener as Events[K]);
      listener(...args);
    };
    return this.on(event, onceListener as Events[K]);
  }
}

// 使用示例：Web3 钱包事件系统
interface WalletEvents {
  'accountChanged': (address: string | null) => void;
  'chainChanged': (chainId: number) => void;
  'connect': (info: { chainId: number }) => void;
  'disconnect': (error: Error) => void;
  'transactionPending': (txHash: string, metadata: { to: string; value: string }) => void;
  'transactionConfirmed': (txHash: string, receipt: TransactionReceipt) => void;
}

const wallet = new TypedEventEmitter<WalletEvents>();

// ✅ 类型正确
wallet.on('accountChanged', (address) => {
  console.log('Address:', address);
});

wallet.emit('chainChanged', 1); // ✅ number 类型
wallet.emit('accountChanged', '0x123...'); // ✅ string 类型

// ❌ 类型错误会报错
// wallet.emit('chainChanged', '1'); // Error: string 不能赋值给 number
// wallet.on('unknownEvent', () => {}); // Error: 事件不存在
```

```typescript
  async emitAsync<K extends keyof Events>(
    event: K,
    ...args: Parameters<Events[K]>
  ): Promise<void> {
    const handlers = this.listeners[event];
    if (!handlers) return;

    // 并行执行所有监听器
    await Promise.all(
      handlers.map(handler => 
        Promise.resolve(handler(...args))
      )
    );
  }
}
```

### 常见问题和解决方案

1. **类型推断失败**：泛型约束不够严格
   - 解决：使用 `extends keyof Events` 约束事件名称

2. **可选参数问题**：某些事件可能没有参数
   - 解决：使用 `void` 作为参数类型

3. **监听器 this 指向**：箭头函数和普通函数的 this 差异
   - 解决：统一使用箭头函数或 bind

## 2. 前端架构演进：从巨石应用到模块化微前端

**考点**: 前端架构、微前端、模块联邦、代码拆分、依赖管理

1. **巨石应用**：所有代码在一个仓库，统一构建部署

- **Module Federation**：Webpack 5 的模块联邦
- **Qiankun**：阿里的微前端框架

## 3. 硬件钱包通信协议：USB HID 与 APDU 指令详解

**考点**: 硬件钱包、USB HID、APDU、通信协议、安全传输

**答案要点**:

### 核心原理解释

硬件钱包通过 USB HID（Human Interface Device）与主机通信，使用 APDU（Application Protocol Data Unit）作为应用层协议。

通信流程：
1. **设备发现**：枚举 USB HID 设备
2. **通道建立**：打开 HID 接口
3. **APDU 封装**：将指令封装为 APDU 格式
4. **分片传输**：大数据分包传输
5. **响应解析**：解析设备返回的 APDU Response

APDU 结构：
- CLA: 指令类别
- INS: 指令码
- P1/P2: 参数
- LC: 数据长度
- Data: 指令数据
- LE: 期望返回长度

### 代码示例

```typescript
// USB HID 通信层
import { Transport } from '@ledgerhq/hw-transport';

class HIDTransport extends Transport {
  private device: HIDDevice;
  
  static async requestDevice(): Promise<HIDDevice> {
    const devices = await navigator.hid.requestDevice({
      filters: [
        { vendorId: 0x2c97 }, // Ledger
        { vendorId: 0x534c }, // Trezor
      ],
    });
    return devices[0];
  }
  
  async exchange(apdu: Buffer): Promise<Buffer> {
    // 封装 APDU
    const wrapped = this.wrapAPDU(apdu);
    
    // 分片发送（HID 报告大小限制 64 字节）
    const chunkSize = 64;
    for (let i = 0; i < wrapped.length; i += chunkSize) {
      const chunk = wrapped.slice(i, i + chunkSize);
      await this.device.sendReport(0, chunk);
    }
    
    // 接收响应
    const response = await this.receiveResponse();
    return this.unwrapAPDU(response);
  }
  
  private wrapAPDU(apdu: Buffer): Buffer {
    // Ledger 的 APDU 封装格式
    const header = Buffer.alloc(5);
    header.writeUInt8(0xe0, 0);  // CLA
    header.writeUInt8(apdu[1], 1); // INS
    header.writeUInt8(apdu[2], 2); // P1
    header.writeUInt8(apdu[3], 3); // P2
    header.writeUInt8(apdu.length - 4, 4); // LC
    
    return Buffer.concat([header, apdu.slice(4)]);
  }
}

// 硬件钱包统一接口
interface HardwareWallet {
  connect(): Promise<string>; // 返回地址

  signTransaction(path: string, tx: Transaction): Promise<string>;
  signMessage(path: string, message: string): Promise<string>;
  signTypedData(path: string, data: TypedData): Promise<string>;
}

// Ledger 实现
class LedgerWallet implements HardwareWallet {
  private transport: Transport;
  private eth: AppEth;
  
  async connect(): Promise<string> {
    this.transport = await TransportWebUSB.create();
    this.eth = new AppEth(this.transport);
    
    // 获取地址（不显示在设备上）
    const { address } = await this.eth.getAddress("44'/60'/0'/0/0");
    return address;
  }
  
  async signTransaction(path: string, tx: Transaction): Promise<string> {
    // 序列化交易
    const unsignedTx = tx.unsignedSerialized.slice(2); // 移除 0x
    
    // 调用 Ledger 签名
    const signature = await this.eth.signTransaction(
      path,
      unsignedTx,
      null, // 不解析链 ID
      null, // 不解析交易类型
      null  // 不解析 nonce
    );
    
    // 组合签名
    const v = parseInt(signature.v, 16);
    const r = '0x' + signature.r;
    const s = '0x' + signature.s;
    
    return Transaction.from({
      ...tx,
      signature: { v, r, s }
    }).serialized;
  }
}
```

## 4. 多链交易解析：EVM、Solana、TON 的差异与抽象

**考点**: 多链架构、交易解析、ABI 解码、数据序列化、链差异处理

**答案要点**:

### 核心原理解释

不同区块链的交易结构差异巨大：

| 特性 | EVM | Solana | TON |
|------|-----|--------|-----|
| 账户模型 | 账户模型 | 账户模型 | 账户模型（无限分片）|
| 交易结构 | RLP 编码 | 二进制（bincode）| Bag of Cells（BoC）|
| 指令模型 | 单一调用 | 多指令数组 | 多动作消息 |
| Gas 模型 | Gas Price × Limit | 计算单元 + 租金 | Gas + 存储费 |
| 地址格式 | 0x...（20字节）| Base58（32字节）| Base64（可变）|

**抽象策略**：
1. **统一数据结构**：定义链无关的交易表示
2. **链特定解析器**：针对每条链实现解析逻辑
3. **风险检测层**：统一的危险操作识别

### 代码示例

```typescript
// 统一交易数据结构
interface ParsedTransaction {
  chainType: 'evm' | 'solana' | 'ton' | 'aptos';
  hash: string;
  from: string;
  to: string;
  value: BigNumber;
  
  // 解析后的数据
  parsed: {
    type: 'transfer' | 'contract_call' | 'token_transfer' | 'swap';
    method?: string;
    params: Record<string, any>;
    
    // 资产变动
    balanceChanges: Array<{
      type: 'native' | 'token' | 'nft';
      symbol: string;
      amount: string;
      direction: 'in' | 'out';
      contract?: string;
    }>;
    
    // 风险标记
    riskFlags: RiskFlag[];
  };
  
  // Gas 信息
  gas: {
    limit: BigNumber;
    used?: BigNumber;
    price: BigNumber;
    totalCost: BigNumber;
  };
}

// EVM 交易解析器
class EVMTransactionParser {
  private abiDecoder = new AbiDecoder();
  
  parse(tx: EthersTransaction, receipt?: TransactionReceipt): ParsedTransaction {
    // RLP 解码已在 ethers 中完成
    
    // 解析合约调用
    let parsedData = this.parseContractData(tx.data, tx.to);
    
    // 识别资产变动
    const balanceChanges = this.extractBalanceChanges(tx, receipt);
    
    // 风险评估
    const riskFlags = this.assessRisk(parsedData, tx, receipt);
    
    return {
      chainType: 'evm',
      hash: tx.hash,
      from: tx.from,
      to: tx.to!,
      value: tx.value,
      parsed: {
        type: parsedData.type,
        method: parsedData.method,
        params: parsedData.params,
        balanceChanges,
        riskFlags,
      },
      gas: {
        limit: tx.gasLimit,
        used: receipt?.gasUsed,
        price: tx.gasPrice || tx.maxFeePerGas!,
        totalCost: receipt ? receipt.gasUsed.mul(tx.effectiveGasPrice!) : BigNumber.from(0),
      },
    };
  }
  
  private parseContractData(data: string, to: string): ParsedData {
    if (!data || data === '0x') {
      return { type: 'transfer', params: {}, method: undefined };
    }
    
    const methodId = data.slice(0, 10);
    
    // 常见方法签名映射
    const signatures: Record<string, { name: string; type: string }> = {
      '0xa9059cbb': { name: 'transfer', type: 'token_transfer' },
      '0x23b872dd': { name: 'transferFrom', type: 'token_transfer' },
      '0x095ea7b3': { name: 'approve', type: 'approval' },
      '0x38ed1739': { name: 'swapExactTokensForTokens', type: 'swap' },
    };
    
    const sig = signatures[methodId];
    if (sig) {
      const abi = this.getABI(to);
      const decoded = abi ? this.abiDecoder.decode(abi, data) : this.decodeParams(data);
      
      return {
        type: sig.type as any,
        method: sig.name,
        params: decoded,
      };
    }
    
    return { type: 'contract_call', method: 'unknown', params: { data } };
  }
}

// Solana 交易解析器
class SolanaTransactionParser {
  parse(tx: SolanaTransaction): ParsedTransaction {
    // 解析指令
    const instructions = tx.instructions.map(ix => {
      const program = this.identifyProgram(ix.programId);
      return {
        program,
        data: this.decodeInstruction(program, ix.data),
        accounts: ix.keys.map(k => ({
          pubkey: k.pubkey.toBase58(),
          isSigner: k.isSigner,
          isWritable: k.isWritable,
        })),
      };
    });
    
    // Solana 特殊：识别 Token 转账
    const tokenTransfers = instructions
      .filter(ix => ix.program === 'token')
      .map(ix => this.parseTokenInstruction(ix));
    
    return {
      chainType: 'solana',
      hash: tx.signatures[0],
      from: tx.feePayer?.toBase58() || '',
      to: instructions[0]?.accounts[0]?.pubkey || '',
      value: BigNumber.from(0),
      parsed: {
        type: instructions.length > 1 ? 'batch' : 'transfer',
        method: instructions.map(i => i.program).join(','),
        params: { instructions },
        balanceChanges: [...tokenTransfers],
        riskFlags: this.assessSolanaRisk(instructions),
      },
      gas: {
        limit: BigNumber.from(0),
        used: BigNumber.from(0),
        price: BigNumber.from(5000), // 5000 lamports per signature
        totalCost: BigNumber.from(tx.signatures.length * 5000),
      },
    };
  }
  
  private identifyProgram(programId: PublicKey): string {
    const known: Record<string, string> = {
      '11111111111111111111111111111111': 'system',
      'TokenkegQfeZyiNwAJbNbGKPFXCWuBvf9Ss623VQ5DA': 'token',
      'ATokenGPvbdGVxr1b2hvZbsiqW5xWH25efTNsLJA8knL': 'associated_token',
      'JUP6LkbZbjS1jKKwapdHNy74zcZ3tLUZoi5QNyVTaV4': 'jupiter', // DEX
    };
    return known[programId.toBase58()] || 'unknown';
  }
}

// 统一解析器
class TransactionParserFactory {
  private parsers: Map<string, TransactionParser> = new Map([
    ['evm', new EVMTransactionParser()],
    ['solana', new SolanaTransactionParser()],
    ['ton', new TONTransactionParser()],
  ]);
  
  parse(chain: string, tx: any): ParsedTransaction {
    const parser = this.parsers.get(chain);
    if (!parser) throw new Error(`Unsupported chain: ${chain}`);
    return parser.parse(tx);
  }
}
```

## 5. EIP-4337 账户抽象：智能合约钱包的设计与实现

**考点**: EIP-4337、账户抽象、UserOperation、Paymaster、Bundler

**答案要点**:

### 核心原理解释

EIP-4337 引入了智能合约钱包的标准，无需修改以太坊协议即可实现账户抽象。

核心组件：
1. **Smart Contract Wallet**：由代码控制的账户
2. **EntryPoint**：统一的入口合约
3. **UserOperation**：代替交易的结构化数据
4. **Bundler**：收集并打包 UserOperations 的节点
5. **Paymaster**：代付 Gas 的合约
6. **Account Factory**：创建智能合约钱包的工厂

关键优势：
- 社交恢复（Social Recovery）
- 多签控制
- 自定义签名算法（如 Passkey）
- Gas 抽象（Paymaster 代付）

```solidity
        address newOwner;
        uint256 approveCount;
        mapping(address => bool) approvals;
        uint256 requestTime;
    }
    RecoveryRequest public recovery;
    
    modifier onlyOwner() {
        require(msg.sender == owner, "Not owner");
        _;
    }

        entryPoint = _entryPoint;
        owner = _owner;
    }

        return entryPoint;
    }

    function _validateSignature(
        UserOperation calldata userOp,
        bytes32 userOpHash
    ) internal override returns (uint256 validationData) {
        // 恢复签名者地址
        bytes32 hash = userOpHash.toEthSignedMessageHash();
        address signer = hash.recover(userOp.signature);
        
        if (signer != owner) {
            return SIG_VALIDATION_FAILED;
        }
        
        return SIG_VALIDATION_SUCCESS;
    }
    
    // 执行交易
    function _execute(address target, uint256 value, bytes calldata data) internal {
        (bool success, bytes memory result) = target.call{value: value}(data);
        require(success, string(result));
    }
    
    // 社交恢复功能
    function initiateRecovery(address newOwner) external {
        require(guardians[msg.sender], "Not guardian");
        require(recovery.requestTime == 0, "Recovery in progress");
        
        recovery.newOwner = newOwner;
        recovery.requestTime = block.timestamp;
        recovery.approveCount = 1;
        recovery.approvals[msg.sender] = true;
    }
    
    function approveRecovery() external {
        require(guardians[msg.sender], "Not guardian");
        require(recovery.requestTime > 0, "No recovery in progress");
        require(!recovery.approvals[msg.sender], "Already approved");
        require(
            block.timestamp < recovery.requestTime + 2 days,
            "Recovery expired"
        );
        
        recovery.approvals[msg.sender] = true;
        recovery.approveCount++;
        
        // 假设需要 3 个 guardian 中的 2 个
        if (recovery.approveCount >= 2) {
            owner = recovery.newOwner;
            delete recovery;
        }
    }
    
    // 添加/移除 guardian
    function addGuardian(address guardian) external onlyOwner {
        guardians[guardian] = true;

interface UserOperation {
  sender: string;          // 智能合约钱包地址
  nonce: BigNumber;        // 防重放
  initCode: string;        // 初始化代码（首次创建）
  callData: string;        // 实际执行的数据
  callGasLimit: BigNumber;
  verificationGasLimit: BigNumber;
  preVerificationGas: BigNumber;
  maxFeePerGas: BigNumber;
  maxPriorityFeePerGas: BigNumber;
  paymasterAndData: string; // Paymaster 信息
  signature: string;       // 签名
}

async function createUserOp(
  wallet: SmartWallet,
  target: string,
  data: string,
  paymaster?: PaymasterAPI
): Promise<UserOperation> {
  const nonce = await wallet.getNonce();
  
  // 构造 callData
  const callData = wallet.interface.encodeFunctionData('execute', [
    target,
    0,
    data,
  ]);
  
  // 估算 Gas
  const gasEstimate = await bundler.estimateUserOpGas({
    sender: wallet.address,
    nonce,
    initCode: '0x',
    callData,
  });
  
  // 构造 UserOperation
  let userOp: UserOperation = {
    sender: wallet.address,
    nonce,
    initCode: '0x',
    callData,
    callGasLimit: gasEstimate.callGasLimit,
    verificationGasLimit: gasEstimate.verificationGasLimit,
    preVerificationGas: gasEstimate.preVerificationGas,
    maxFeePerGas: await provider.getFeeData().then(f => f.maxFeePerGas!),
    maxPriorityFeePerGas: await provider.getFeeData().then(f => f.maxPriorityFeePerGas!),
    paymasterAndData: '0x',
    signature: '0x',
  };
  
  // 如果有 Paymaster，获取赞助数据
  if (paymaster) {
    userOp.paymasterAndData = await paymaster.getPaymasterData(userOp);
  }
  
  // 签名
  const opHash = await entryPoint.getUserOpHash(userOp);
  userOp.signature = await owner.signMessage(arrayify(opHash));
  
  return userOp;
}
```

```typescript
// 合约安全扫描器
class ContractSecurityScanner {
  private riskDB: RiskDatabase;
  private bytecodeAnalyzer: BytecodeAnalyzer;
  
  async scanContract(address: string): Promise<SecurityReport> {
    const [code, source] = await Promise.all([
      this.getBytecode(address),
      this.getSourceCode(address),
    ]);
    
    const risks: RiskFinding[] = [];
    
    // 1. 已知恶意合约检查
    const knownRisk = await this.riskDB.checkAddress(address);
    if (knownRisk) {
      risks.push({
        severity: 'critical',
        category: 'known_malicious',
        description: knownRisk.description,
        source: knownRisk.source,
      });
    }
    
    // 2. 字节码分析
    const byteAnalysis = this.bytecodeAnalyzer.analyze(code);
    
    // 检查自毁函数
    if (byteAnalysis.opcodes.includes('SELFDESTRUCT')) {
      risks.push({
        severity: 'high',
        category: 'selfdestruct',
        description: '合约包含自毁功能，可能被销毁',
      });
    }
    
    // 检查未检查的外部调用
    if (byteAnalysis.hasUncheckedCall) {
      risks.push({
        severity: 'high',
        category: 'unchecked_call',
        description: '存在未检查返回值的外部调用，可能被重入攻击',
      });
    }
    
    // 3. 蜜罐检测
    const isHoneypot = await this.detectHoneypot(address);
    if (isHoneypot) {
      risks.push({
        severity: 'critical',
        category: 'honeypot',
        description: '疑似蜜罐合约，买入后可能无法卖出',
      });
    }
    
    // 4. 模拟执行检测
    const simResult = await this.simulateInteractions(address);
    if (simResult.suspicious) {
      risks.push(...simResult.findings);
    }
    
    return {
      address,
      overallRisk: this.calculateOverallRisk(risks),
      risks,
      timestamp: Date.now(),
    };
  }
  
  // 蜜罐检测：模拟买入卖出
  private async detectHoneypot(tokenAddress: string): Promise<boolean> {
    const forkProvider = await this.createFork();
    const tester = new ContractTester(forkProvider);
    
    try {
      // 模拟买入
      const buyResult = await tester.simulateBuy(tokenAddress, '1');
      if (!buyResult.success) return false; // 连买都不让
      
      // 模拟卖出（延迟检查，有些蜜罐有时间锁）
      await tester.advanceTime(60); // 前进 1 分钟
      
      const sellResult = await tester.simulateSell(tokenAddress, buyResult.tokensReceived);
      if (!sellResult.success) {
        return true; // 能买不能卖 = 蜜罐
      }
      
      // 检查滑点异常
      const slippage = calculateSlippage(buyResult, sellResult);
      if (slippage > 99) {
        return true; // 卖出损失过大
      }
      
      return false;
    } finally {
      await forkProvider.destroy();
    }
  }
  
  // 钓鱼网站检测
  async checkPhishingSite(url: string): Promise<PhishingReport> {
    const domain = new URL(url).hostname;
    
    // 1. 黑名单检查
    if (await this.isBlacklisted(domain)) {
      return { isPhishing: true, confidence: 1, reason: 'Domain blacklisted' };
    }
    
    // 2. 相似域名检测
    const similar = await this.findSimilarDomains(domain);
    if (similar.some(s => s.similarity > 0.9)) {
      return {
        isPhishing: true,
        confidence: 0.9,
        reason: `Domain similar to ${similar[0].target}`,
      };
    }
    
    // 3. SSL 证书检查
    const cert = await this.checkCertificate(domain);
    if (!cert.valid || cert.age < 7) {
      return {
        isPhishing: true,
        confidence: 0.6,
        reason: 'Suspicious SSL certificate',
      };
    }
    
    // 4. 页面内容分析
    const content = await this.fetchPageContent(url);
    const indicators = this.analyzePhishingIndicators(content);
    
    return {
      isPhishing: indicators.score > 0.7,
      confidence: indicators.score,
      reason: indicators.reasons.join(', '),
    };
  }
```

```typescript
class SecureExecutor {
  private policy: Policy;
  
  async execute(intent: Intent): Promise<Result> {
    // 1. 检查权限
    if (!this.policy.allows(intent)) {
      return { error: 'Not allowed' };
    }
```
